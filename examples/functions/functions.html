Starlight Functions Guide

Learn how to define and use functions in Starlight

1. Function Declaration


# Define a function
func greet() {

    # Print message
    sldeploy "Hello!";
}

# Call function
greet();

2. Function with Parameters


# Function with parameters
func greet(name) {

    # Use parameter
    sldeploy "Hello " + name;
}

greet("Alice");
greet("Bob");

3. Return Values


# Function returning value
func add(a, b) {

    # Return result
    return a + b;
}

let result = add(2, 3);

sldeploy "Result: " + result;

4. Default Behavior (No Return)


# Function without return
func test() {

    sldeploy "Running function";
}

# Returns null automatically
let value = test();

sldeploy "Returned: " + value;

5. Arrow Functions


# Short arrow function
let add = (a, b) => a + b;

let result = add(5, 10);

sldeploy result;

6. Arrow Function (Block Form)


# Arrow with multiple lines
let multiply = (a, b) => {

    # Explicit return
    return a * b;
};

sldeploy multiply(3, 4);

7. Functions Inside Functions


# Outer function
func outer() {

    # Inner function
    func inner() {
        sldeploy "Inside inner function";
    }

    inner();
}

outer();

8. Function with Condition


# Function with logic
func check(num) {

    if (num > 0) {
        return "Positive";
    }

    return "Negative";
}

sldeploy check(10);
sldeploy check(-5);

9. Passing Function as Value


# Function that accepts another function
func apply(fn, value) {

    return fn(value);
}

# Arrow function as argument
let result = apply((x) => x * 2, 5);

sldeploy result;

10. Recursive Function


# Recursive factorial
func factorial(n) {

    # Base case
    if (n == 0) {
        return 1;
    }

    # Recursive call
    return n * factorial(n - 1);
}

sldeploy factorial(5);
