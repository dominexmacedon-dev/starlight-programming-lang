Starlight Objects Guide

Learn how to work with objects and structured data in Starlight

1. Creating Objects


# Create an object
let user = {
    "name": "Alice",
    "age": 25
};

# Print object
sldeploy user.name;

2. Accessing Properties


# Access using dot notation
sldeploy user.name;

# Access using bracket notation
sldeploy user["age"];

3. Updating Properties


# Update existing value
user.age = 30;

# Add new property
user.city = "New York";

sldeploy user;

4. Loop Through Object


# Iterate over keys
for key in user {

    # Access value using key
    sldeploy key + ": " + user[key];
}

5. Nested Objects


# Create nested object
let person = {
    "name": "Bob",
    "address": {
        "city": "London",
        "zip": 12345
    }
};

# Access nested value
sldeploy person.address.city;

6. Objects Inside Arrays


# Array of objects
let users = [
    { "name": "Alice", "age": 25 },
    { "name": "Bob", "age": 30 }
];

# Loop through array
for item in users {

    # Access object properties
    sldeploy item.name + " - " + item.age;
}

7. Check Property Exists


# Check if key exists
if (has(user, "name")) {
    sldeploy "Name exists";
}

8. Merge Objects


# Merge two objects
let a = { "x": 1 };
let b = { "y": 2 };

let merged = merge(a, b);

sldeploy merged;

9. Clone Object


# Create copy
let copy = clone(user);

# Modify copy
copy.name = "Charlie";

# Original remains unchanged
sldeploy user.name;
sldeploy copy.name;

10. Real Example (User System)


# Store multiple users
let users = {};

# Add user
users["alice"] = { "balance": 100 };

# Access user
let u = users["alice"];

# Update balance
u.balance = u.balance + 50;

sldeploy "Balance: " + u.balance;
