Starlight Looping Guide

Learn how loops work in the Starlight programming language

1. While Loop


# Initialize counter
define i = 0;

# Loop runs while condition is true
while (i < 5) {

    # Print current value
    sldeploy "i: " + i;

    # Increase counter
    i = i + 1;
}

2. For Loop


# Loop with initialization, condition, increment
for (define i = 0; i < 5; i = i + 1) {

    # Print current iteration
    sldeploy "i: " + i;
}

3. For-In Loop (Array)


# Create an array
define arr = [10, 20, 30];

# Iterate over each element
for item in arr {

    # Print each value
    sldeploy "Value: " + item;
}

4. Break Statement


# Loop with break condition
for (define i = 0; i < 10; i = i + 1) {

    # Stop loop when i equals 5
    if (i == 5) {
        break;
    }

    # Print values before break
    sldeploy "i: " + i;
}

5. Continue Statement


# Loop with continue
for (define i = 0; i < 5; i = i + 1) {

    # Skip when i equals 2
    if (i == 2) {
        continue;
    }

    # Print remaining values
    sldeploy "i: " + i;
}
